 CREATE OR REPLACE PROCEDURE EMPL_EXP(EMPID IN NUMBER,EMPNAME OUT VARCHAR,EXP OUT NUMBER)
    IS
    TODAYS DATE;
    DOJ DATE;
    BEGIN
    SELECT SYSDATE INTO TODAYS FROM DUAL;
    SELECT HIRE_DATE INTO DOJ FROM EMPLOYEE WHERE EMPLOYEE_ID = EMPID;
    EXP:=(TODAYS-DOJ)/365;
    SELECT FIRST_NAME INTO EMPNAME FROM EMPLOYEE WHERE EMPLOYEE_ID=EMPID;
   END;
   /
